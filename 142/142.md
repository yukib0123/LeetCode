/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if (head == NULL || head->next == NULL) return nullptr;
        ListNode *fast = head->next->next;
        ListNode *slow = head->next;
        while (fast != slow) {
            if (fast == NULL || fast->next == NULL || fast->next->next == NULL) return nullptr;
            fast = fast->next->next;
            slow = slow->next;
        }
        ListNode *start = head;
        while (start != slow) {
            start = start->next;
            slow = slow->next;
        }
        return slow;
    }
};
